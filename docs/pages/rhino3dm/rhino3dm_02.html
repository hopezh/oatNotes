        # if geom is PolylineCurve, print its details
        if isinstance(obj.Geometry, rhino3dm.PolylineCurve):

            print('            curve IsPolyline   :', obj.Geometry.IsPolyline())

            print('            curve SegmentCount :', obj.Geometry.ToPolyline().SegmentCount)
